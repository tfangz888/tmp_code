import pandas as pd

def sort_row(row):
  c = d = None
  a = row[row.notna()] # 过滤掉NaN
  longBuy = a[a<0] # 只要负收益
  longBuy.index  = [(x, -1) for x in longBuy.index]
  a = np.negative(a) # short
  shortSell = a[a<0] # 只要负收益
  shortSell.index  = [(x, 1) for x in shortSell.index]
  ret = longBuy.append(shortSell)
  ret = ret.sort_values()
  return list(ret.index)
